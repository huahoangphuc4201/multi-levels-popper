```bash
npm install --save multi-levels-popper
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'multi-levels-popper'
import 'multi-levels-popper/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
